Linux 驱动开发主要分为3大类

 1. 字符设备驱动，最多的。
 2. 块设备驱动，存储。
 3. 网络设备驱动。

什么是驱动：

驱动的本质就是 获取外设、传感器的原始数据，控制外设。 数据会提交到应用程序。

Linux下驱动编写既要编写一个驱动还要编写一个简单的测试程序，APP。单片机下的驱动和应用都是在一个文件也可以说在一个整体下，获取到数据直接使用，但是Linux下的驱动是一个文件，共别人去调用。相当于封装为系统的API接口。

用户空间和内核空间：

应用程序都运行在用户空间，驱动一类都运行在内核空间。 用户空间就是为了避免不正确的使用底层。造成系统错误。

应用程序访问内核的方法：

系统调用，异常（中断）和陷入。

应用程序不会直接访问内核，而实通过API函数来间接访问内核。例如，用户调用print函数，内核执行write函数。每个系统调用都有一个系统调用号。系统调用处于内核空间，应用程序无法直接访问，需要“陷入”到内核，方法就是软中断，陷入内核后还要指定系统调用号。最终指向到硬件平台。

字符设备驱动流程：

​ 1。 Linux里面一切皆文件，驱动设备表现就是，在/dev/目录下的一个文件，，/dev/led/.

我们调用它，应用程序通过write写 1 led梁，写 0 led灭。

​ 驱动是被应用程序调用的。

​ 驱动是分框架的。

字符设备驱动框架：

​ 字符设备驱动编写主要就是驱动对应的open, write, close 等。。。其实就是一个大的结构体。

驱动模块的加载与卸载：

Linux驱动程序可以编译到kernel 中， 也就是zImage， 也可以编译为模块，.ko。

Linux驱动编写需要内核代码，所以最好要有移植好的三巨头代码。方便很多。

内核代码是没有printf();的 有的是printk();

Linux 设备号：

Linux将设备号分为2部分， 主设备号和次设备号

主设备号占用前12为，此设备号占用低20位。

设备号操作函数|宏 ：

从dev_t 获取主设备号和此设备号 MAJOR（dev_t） MINOR(dev_t).

也可以使用主设备号和此设备号构成dev_t， 通过MKDEV(major, minor);

应用程序编写：

加载驱动。

modprobe chadevbase c 200 0

测试

内存映射：

 1. 裸机LED灯实验就是操作6ULL的寄存器。 驱动开发也可以直接操作寄存器。
 2. Linux下驱动开发也可以操作寄存器，Linux 不能直接对寄存器物理地址进行读写操作。比如寄存器A物理地址为0x01010101, 裸机的时候可以直接对0x0101010101 这个物理地址进行操作，但是Linux不行，因为Linux会使能MMU。

加载驱动：

使用 insmod chrdevbase.ko 或 modprobe chrdevbase.ko 加载驱动文件：

加载失败的话 试试执行 depmod 会自动生成几个文件 modules.alias、modules.symbols 和 modules.dep

再次执行 insmod chrdevbase.ko 或 modprobe chrdevbase.ko 加载驱动文件：

加载成功之后。对应命令：

1. depmod 			 // 第一次加载驱动必须的
2. modprobe led.ko	 // 加载驱动 会执行 init 函数
3. ls /proc/devices  // 会看到 设备号和设备名  例如 设备号为 200  设备名为 led
3. rmmod led.ko		 // 卸载驱动

驱动一个简单外设步骤：

cortexA 系列的 start.s 文件一般是用来初始化一些底层的寄存器，之后设置 sp 指针跳入到C语言环境。

编写一个驱动步骤：

1.需要用汇编初始化一些外设，

2.使用汇编初始化DDR。

3设置sp指针一般指向DDR。

4.设置好C语言指向环境。 由于C语言的执行需要入栈出栈，所以要设置好sp指针。

5.CCM_CCGR【0 - 6】 7 个寄存器控制所有外设时钟使能。

6.IO复用寄存器 IOMUXC_SW_MUX_CTL_PAD_GPIO1_IO03 LED0 引脚 P1_3 将bit0~3 设置为 5，复用为普通IO口。

7，设置电气属性 IOMUXC_SW_PAD_CTL_PAD_GPIO1_IO03

点灯代码：

LED.c

#include <linux/types.h>
#include <linux/kernel.h>
#include <linux/delay.h>
#include <linux/ide.h>
#include <linux/init.h>
#include <linux/module.h>
#include <linux/uaccess.h>

/*
注册驱动的注册和卸载
*/

// 初始化一些SOC外设， 初始化DDR外设（这一步IMX6U不需要），
// 设置Sp指针指向DDR，初始化C语言环境。 程序入栈出栈需要SP指针。

// 主设备号
#define LED_MAJOR 200

// 设备名
#define LED_NAME "led"

// 时钟使能     CCM_CCGR1 的基地址                 Address: 20C_4000h base + 6Ch offset = 20C_406Ch
#define CCM_CCGR1_BASE (0X020C406C)

// IO复用       IOMUXC_SW_MUX_CTL_PAD_GPIO1_IO03  Address: 20E_0000h base + 68h offset = 20E_0068h
#define SW_MUX_GPIO1_IO03_BASE (0X020E0068)

// 设置电气属性 IOMUXC_SW_MUX_CTL_PAD_LCD_DATA03  Address: 20E_0000h base + 124h offset = 20E_0124h
#define SW_PAD_GPIO1_IO03_BASE (0X020E02F4)

// 控制高低电平 GPIOx_DR                          Address: 209_C000（Base address） + 0h offset
#define GPIO1_DR_BASE (0X0209C000)

// 设置输入输出模式 GPIOx_GDIR                    Address: 209_C000（Base address） + 4h offset
#define GPIO1_GDIR_BASE (0X0209C004)

// 物理地址映射后虚拟地址指针
static void __iomem *IMX6U_CCM_CCGR1;       // 时钟
static void __iomem *IMX6SW_MUX_GPIO1_IO03; // IO复用
static void __iomem *SW_PAD_GPIO1_IO03;     // 电器属性
static void __iomem *GPIO1_DR;              // 输出电平
static void __iomem *GPIO1_GDIR;            // 输入输出模式

#define LEDOFF 0 // LED 灭
#define LEDON 1  // LED亮

void led_Switch(u8 sta)
{
    u32 val = 0;
    if (sta == LEDON) // open led
    {
        val = readl(GPIO1_DR);
        val &= ~(1 << 3); // 清零 打开LED灯。
        writel(val, GPIO1_DR);
    }
    else if (sta == LEDOFF)
    {
        u32 val = readl(GPIO1_DR);
        val |= (1 << 3);
        writel(val, GPIO1_DR);
    }
}

/*  open 的函数原型
    int (*open)(struct inode *, struct file *);
*/
static int led_open(struct inode *inode, struct file *filp)
{
    return 0;
}

/*
    release 的函数原型
    int (*release) (struct inode *, struct file *);
*/
static int led_release(struct inode *inode, struct file *file)
{
    return 0;
}

/*
    write 的函数原型
    ssize_t (*write)(struct file *, const char __user *, size_t, loff_t *);
*/
static ssize_t led_write(struct file *filp, const char __user *buf, size_t count, loff_t *ppos)
{

    int retvalue;
    u8 databuf[1];
    retvalue = copy_from_user(databuf, buf, count);
    if (retvalue < 0)
    {
        printk("kernel write failed! \r\n");
        return -EFAULT;
    }

    // 判断开灯还是关灯
    led_Switch(databuf[0]);

    return 0;
}

static const struct file_operations led_fops = {

    .owner = THIS_MODULE,
    .open = led_open,
    .write = led_write,
    .release = led_release,

};

/* 驱动加载入口 */
static int __init led_init(void)
{
    // 1. 初始化LED灯。
    // #define ioremap(cookie,size)		__arm_ioremap((cookie), (size), MT_DEVICE)
    // extern void __iomem *__arm_ioremap(phys_addr_t, size_t, unsigned int);
    // ioremap 就是将物理地址映射为虚拟地址。
    // // phys_addr_t 要映射的物理地址 size_t 要映射资源的大小  32bit = 4Byte。
    IMX6U_CCM_CCGR1 = ioremap(CCM_CCGR1_BASE, 4);

    // 映射IO复用寄存器地址
    IMX6SW_MUX_GPIO1_IO03 = ioremap(SW_MUX_GPIO1_IO03_BASE, 4);

    // 映射电器属性寄存器地址
    SW_PAD_GPIO1_IO03 = ioremap(SW_PAD_GPIO1_IO03_BASE, 4);

    // 映射输出电平寄存器地址
    GPIO1_DR = ioremap(GPIO1_DR_BASE, 4);

    // 映射输入输出模式寄存器地址
    GPIO1_GDIR = ioremap(GPIO1_GDIR_BASE, 4);

    // 初始化
    u32 val = 0;
    val = readl(IMX6U_CCM_CCGR1);
    val &= ~(3 << 26); // 先清除配置的bit26，27
    val |= 3 << 26;    // 将26， 7 置1
    writel(val, IMX6U_CCM_CCGR1);

    // 设置复用
    writel(0x5, IMX6SW_MUX_GPIO1_IO03);

    // 设置电气属性
    writel(0x10B0, SW_PAD_GPIO1_IO03);

    // 设置GPIO1_3 为输出模式
    val = readl(GPIO1_GDIR);
    val |= 1 << 3; // 将 bit3 置1
    writel(val, GPIO1_GDIR);

    // 点亮
    val = readl(GPIO1_DR);
    val &= (1 << 3); // 清零 打开LED灯。
    writel(val, GPIO1_DR);

    int ret;
    // 1. 注册字符设备
    ret = register_chrdev(LED_MAJOR, LED_NAME, &led_fops); // 注册设备
    if (ret < 0)
    {
        printk("led Init Error\r\n");
        return EIO; // IO error
    }
    printk("LED驱动注册\r\n");
    return 0;
}

static void __exit led_exit(void)
{

    u32 val = readl(GPIO1_DR);
    val |= (1 << 3);
    writel(val, GPIO1_DR);
    printk("关闭LED灯\n");

    // 取消虚拟地址映射
    iounmap(IMX6U_CCM_CCGR1);
    iounmap(IMX6SW_MUX_GPIO1_IO03);
    iounmap(SW_PAD_GPIO1_IO03);
    iounmap(GPIO1_DR);
    iounmap(GPIO1_GDIR);

    unregister_chrdev(LED_MAJOR, LED_NAME); // 注销设备
    printk("LED驱动卸载\r\n");
}

module_init(led_init);
module_exit(led_exit);

MODULE_LICENSE("GPL");
MODULE_AUTHOR("Wyt");

LEDAPP.c

#include <sys/types.h>
#include <sys/stat.h>
#include <fcntl.h>
#include <stdio.h>
#include <unistd.h>
#include <string.h>

/*
    int argc : 参数个数
    char *argv[] : 参数内容  1 开灯    0 关灯
 */
#define LEDOFF 0
#define LEDON 1
int main(int args, char **argv)
{
    int fd;
    int ret;
    char *filename;
    unsigned char databuf[1];

    if (args != 3)
    {
        printf("请确认传入的参数是否正确\n");
        return -1;
    }

    filename = argv[1];

    fd = open(filename, O_RDWR);
    if (fd < 0)
    {
        printf(" LED 打开失败\n");
        return -1;
    }

    // 将字符转为数字
    databuf[0] = atoi(argv[2]);

    ret = write(fd, databuf, sizeof(databuf));
    if (ret < 0)
    {
        printf("LED控制失败\n");
        close(fd);
        return -ret;
    }

    close(fd);

    return 0;
}

  3. 在开发板挂载，

  4. modprobe led.ko

  5. mknod /dev/led c 200 0

  6. ./ledApp /dev/led 1 亮

  7. ./ledApp / dev/led 0 灭
