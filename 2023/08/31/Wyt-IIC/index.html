#define SDA_IN()  {GPIOA->CRL&=0X0FFFFFFF;GPIOA->CRL|=(u32)8<<28;}
#define SDA_OUT() {GPIOA->CRL&=0X0FFFFFFF;GPIOA->CRL|=(u32)3<<28;}

上面的意思, 其实就是切换PA_7的模式, 控制模式的寄存器叫CRL跟CRH,

CRL(引脚0-7):

CRH(引脚8-16):

​ 也就是每个引脚由4个位控制，

GPIOA->CRL&=0X0FFFFFFF; 既然要配置，那么就要先清零，所以第一步写入：GPIOB->CRL&=0x0FFFFFFF;将寄存器设置为0000 1111 1111 1111 1111 1111 1111 1111；

GPIOA->CRL|=(u32)8<<28; 第一步配置输入：图中可得输入格式写入00；第二步配置上拉：图中可得上拉格式写入10（下拉也是10），所以写入的四位就是10和00，对应就是1000，16进制就是8，所以写入GPIOB->CRL|=0x80000000;配置完成。与移位的操作是一样的意思。

OUT 也一样。

Wyt_IIC.c

/******************************************************/
/* 王永泰的IIC文件  ---------- 适用于Stm32F103系列  */

#include "iic.h"
#include "delay.h"

// IIC初始化
void Wyt_IIC_Init(void)
{
	GPIO_InitTypeDef GPIO_InitStructure;
	RCC_APB2PeriphClockCmd(RCC_APB2Periph_GPIOB, ENABLE); // 使能GPIOB时钟

	GPIO_InitStructure.GPIO_Pin = GPIO_Pin_6 | GPIO_Pin_7;
	GPIO_InitStructure.GPIO_Mode = GPIO_Mode_Out_PP; // 推挽输出
	GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;
	GPIO_Init(GPIOB, &GPIO_InitStructure);

	GPIO_SetBits(GPIOB, GPIO_Pin_6 | GPIO_Pin_7); // PB6,PB7 输出高
}

// 起始信号  ->    SCL and SDA都为高时，SDA由高变低。即为开始信号，注意发送信号时SCL是高电平，发送数据时SCL时低电平。
void Wyt_IIC_Start(void)
{

	Wyt_SDA_OUT();	 // 将PB7 改为输出模式， 和 init 里的配置是一样的，只不过是用了寄存器来做。
	Wyt_IIC_SCL = 1; // 先将时钟线拉高。
	Wyt_IIC_SDA = 1; // 数据线也拉高。
	delay_us(4);	 // 由于IIC 时序 大概在400KHZ ，所以一个周期为2.5us，高电平时间为1.25us，此时4us 所以此速率时小于400khz的。

	// 之后 时钟线和数据线都为高电平时，现在数据线 由 高电平 -> 低电平， 为开始信号，so

	Wyt_IIC_SDA = 0; // 发送开始信号。
	delay_us(4);
	Wyt_IIC_SCL = 0; // 钳住IIC，通信过程中，只有在时钟线拉低的情况下，数据线才能被操作，所以拉低。
}

// 停止信号  ->    SCL and SDA都为高时，SDA由低变高。即为结束信号，注意发送信号时SCL是高电平，发送数据时SCL时低电平。
void Wyt_IIC_Stop(void)
{

	Wyt_SDA_OUT(); // 将PB7 改为输出模式， 和 init 里的配置是一样的，只不过是用了寄存器来做。

	Wyt_IIC_SCL = 0; // 先将时钟线拉低。
	Wyt_IIC_SDA = 1; // 数据线也拉高。

	delay_us(4); // 由于IIC 时序 大概在400KHZ ，所以一个周期为2.5us，高电平时间为1.25us，此时4us 所以此速率时小于400khz的。

	Wyt_IIC_SCL = 1; // 先将时钟线拉高。
	Wyt_IIC_SDA = 1; // 数据线也拉高。

	delay_us(4); // 由于IIC 时序 大概在400KHZ ，所以一个周期为2.5us，高电平时间为1.25us，此时4us 所以此速率时小于400khz的。
}

// IIC等待应答信号  ->  SDA为输入模式，
u8 Wyt_IIC_Wait_Ask(void)
{
	u8 askTimecount = 0; // 等待 应答信号计数
	Wyt_SDA_IN();		 // 等待信号必然是接受了，

	Wyt_IIC_SDA = 1; // SDA拉高，
	delay_us(2);

	Wyt_IIC_SCL = 1; // SCL拉高，
	delay_us(2);

	while (Wyt_IIC_READ_SDA) // 等待IIC_SDA变为高电平
	{
		askTimecount++;
		if (askTimecount > 250)
		{
			IIC_Stop(); // 发送停止信号， 表明接下来没有数据要传输了，本次传输结束。
			return 1;	// 没有应答
		}
	}
	Wyt_IIC_SCL = 0; // 拉低时钟线。
}

// 发送应答信号，
void Wyt_IIC_Not_ack(void)
{
	Wyt_IIC_SCL = 0; // SCL为低
	Wyt_SDA_OUT();
	Wyt_IIC_SDA = 0; // 此时SDA为低，不应答。
	delay_us(4);
	Wyt_IIC_SCL = 1;
	delay_us(4);
	Wyt_IIC_SCL = 0;
}

void Wyt_IIC_ack(void)
{
	Wyt_IIC_SCL = 0; // SCL 为低。
	Wyt_SDA_OUT();
	Wyt_IIC_SDA = 1; // 此时SDA 为高则 应答。
	delay_us(4);
	Wyt_IIC_SCL = 1;
	delay_us(4);
	Wyt_IIC_SCL = 0;
}

void Wyt_IIC_SendByte(u8 byte)
{
	u8 t = 0;
	Wyt_SDA_OUT();
	Wyt_IIC_SCL = 0; // 开始传输数据
	for (t = 0; t < 8; t++)
	{
		Wyt_IIC_SDA = (byte & 0x80) >> 7; // 得到最高位。
		byte <<= 1;						  // 向左移一位，
		delay_us(2);
		Wyt_IIC_SCL = 1;
		delay_us(2);
		Wyt_IIC_SCL = 0;
		delay_us(2);
	}
}

// IIC读取一个字节
u8 Wyt_IIC_ReadByte(u8 ack)
{
	u8 i, receive = 0;
	Wyt_SDA_IN();
	for (i = 0; i < 8; i++)
	{
		Wyt_IIC_SCL = 0;
		delay_us(2);
		Wyt_IIC_SCL = 1;
		receive <<= 1;
		if (Wyt_IIC_READ_SDA)
		{
			receive++;
		}
		delay_us(1);
	}
	if (ack)
	{
		Wyt_IIC_Not_ack();
	}
	else
	{
		Wyt_IIC_ack();
	}
	return receive;
}

/******************************************************/
/* 王永泰的IIC头文件  ---------- 适用于Stm32F103系列  */

#ifndef __IIC_H
#define __IIC_H

#include "sys.h"

// 配置IIC的引脚输入输出模式。 和初始化 那一步的操作一样， 只不过是用寄存器来操作，快。
#define Wyt_SDA_IN()                \
    {                               \
        GPIOB->CRL &= 0X0FFFFFFF;   \
        GPIOB->CRL |= (u32)8 << 28; \
    }
#define Wyt_SDA_OUT()               \
    {                               \
        GPIOB->CRL &= 0X0FFFFFFF;   \
        GPIOB->CRL |= (u32)3 << 28; \
    }

#define Wyt_IIC_SDA PBout(7)
#define Wyt_IIC_SCL PBout(6)
#define Wyt_IIC_READ_SDA PBin(7)

void Wyt_IIC_Init(void);        /*  IIC引脚初始化  */
void Wyt_IIC_Start(void);       /* IIC起始信号*/
void Wyt_IIC_Stop(void);        /*IIC终止信号*/
u8 Wyt_IIC_Wait_Ask(void);      /* IIC等待应答*/
void Wyt_IIC_ack(void);         /* IIC发送应答*/
void Wyt_IIC_Not_ack(void);     /*    IIC 不发送应答*/
void Wyt_IIC_SendByte(u8 byte); /*IIC发送一个byte*/

#endif
