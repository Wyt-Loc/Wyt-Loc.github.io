让智能车跑动起来

如何让智能车跑起来
发布日期:   2023-03-31
更新日期:   2023-04-26
文章字数:   1.1k
阅读时长:   4 分

最单的跑起来，无非就是电机+舵机+编码器。

通过一些很简单的函数就可以做到跑起来，但是跑起来之后的事情就难了。

还是直接贴代码吧，都是直接去掉用一些龙邱的官方库，

分别是，电机，舵机，编码器的初始化

/*
 * My_Servos.c
 *
 *  Created on: 2023年3月31日
 *      Author: Wyt
 */

#include <My_Servos.h>
#include <LQ_GTM.h>
#include <LQ_Atom_Motor.h>


void Servos_Io_init(void){

    ATOM_PWM_InitConfig(ATOMSERVO2, 1900, 100);

}

void Servos_Set_Angle(void){

    ATOM_PWM_SetDuty(ATOMSERVO2, 1940, 100);

}

/*
 * My_Motor.c
 *
 *  Created on: 2023年3月31日
 *      Author: Wyt
 */

#include <LQ_Atom_Motor.h>
#include <My_Motor.h>
#include <LQ_GTM.h>


/**
 * 电机PWM输出角初始化  刚开始默认电机没有电压 防止直接跑出去
 */

void Mortor_Init(void){

    ATOM_PWM_InitConfig(ATOMPWM0, 0, 12500);
    ATOM_PWM_InitConfig(ATOMPWM1, 0, 12500);

}


/**
 * 电机速度设置
 */
void Morter_Set_V(void){

    ATOM_PWM_SetDuty(ATOMPWM0, 1000, 12500);//驱动两个电机   1000应该是最小的再小不走了
    ATOM_PWM_SetDuty(ATOMPWM1, 1000, 12500);

}
/*
 * My_Encoder.c
 *
 *  Created on: 2023年3月31日
 *      Author: Wyt
 */
#include <LQ_TFT18.h>
#include <My_encoder.h>
#include <LQ_GPT12_ENC.h>
#include <stdio.h>

volatile int encv_left = 0;     //左轮编码器值
volatile int encv_right = 0;    //右轮编码器值
volatile int left_sum = 0;      //左轮总共的距离
volatile int right_sum = 0;     //右轮总共的距离

#define TFT_flag  0

int a = 0;


/**
 *  编码器IO初始化
 */
void Encoder_Io_Init(void){

    ENC_InitConfig(ENC2_InPut_P33_7, ENC2_Dir_P33_6);
    ENC_InitConfig(ENC5_InPut_P10_3, ENC5_Dir_P10_1);


#if TFT_flag

    TFTSPI_Init(1);        //LCD初始化  0:横屏  1：竖屏
    TFTSPI_CLS(u16BLUE);   //蓝色屏幕
    TFTSPI_P8X16Str(0, 0, "Test Encoder",u16WHITE,u16BLACK);      //字符串显示

#endif


}

/**
 * 得到编码器反馈数值
 * 针对不同情况进行处理 当前状态编码器2数值前进为负，编码器4前进为正
 */

void Get_Encoder_Data(void){

    char txt[5];
    encv_left  =  ENC_GetCounter(ENC2_InPut_P33_7);      //注意左轮为负
    encv_right = -ENC_GetCounter(ENC5_InPut_P10_3);      //右轮值正常

#if TFT_flag
    sprintf(txt, "Enc2: %05d; ", encv_left);
    TFTSPI_P8X16Str(0, 2, txt, u16WHITE, u16BLACK);       //字符串显示

    sprintf(txt, "Enc4: %05d; ", encv_right);
    TFTSPI_P8X16Str(0, 3, txt, u16WHITE, u16BLACK);       //字符串显示

    left_sum +=  encv_left;
    right_sum += encv_right;

    //龙邱512带方向编码器1米5790个脉冲    经过测试发现 差不多 半米2900多个脉冲计数
    sprintf(txt, "left:%05d; ", left_sum);
    TFTSPI_P8X16Str(0, 4, txt, u16WHITE, u16BLACK);       //字符串显示

    sprintf(txt, "right: %05d; ", right_sum);
    TFTSPI_P8X16Str(0, 5, txt, u16WHITE, u16BLACK);       //字符串显示
#endif

}

#include <include.h>//各个模块的头文件
#include <IfxCpu.h>
#include <IfxScuCcu.h>
#include <IfxScuWdt.h>
#include <IfxStm.h>
#include <IfxStm_reg.h>
#include <LQ_GPIO_LED.h>
#include <LQ_TFT18.h>
#include <LQ_UART.h>
#include <Main.h>
#include <My_encoder.h>
#include <My_Motor.h>
#include <My_Servos.h>


App_Cpu0 g_AppCpu0; // brief CPU 0 global data
IfxCpu_mutexLock mutexCpu0InitIsOk = 1;   // CPU0 初始化完成标志位
volatile char mutexCpu0TFTIsOk=0;         // CPU1 0占用/  1释放 TFT

void CPU_Init(void){

    // 关闭CPU总中断
    IfxCpu_disableInterrupts();
    // 关闭看门狗，如果不设置看门狗喂狗需要关闭
    IfxScuWdt_disableCpuWatchdog(IfxScuWdt_getCpuWatchdogPassword());
    IfxScuWdt_disableSafetyWatchdog(IfxScuWdt_getSafetyWatchdogPassword());
    // 读取总线频率
    g_AppCpu0.info.pllFreq = IfxScuCcu_getPllFrequency();
    g_AppCpu0.info.cpuFreq = IfxScuCcu_getCpuFrequency(IfxCpu_getCoreIndex());
    g_AppCpu0.info.sysFreq = IfxScuCcu_getSpbFrequency();
    g_AppCpu0.info.stmFreq = IfxStm_getFrequency(&MODULE_STM0);
    // 开启CPU总中断
    IfxCpu_enableInterrupts();

}

/*************************************************************************
*  函数名称：int core0_main (void)
*  功能说明：CPU0主函数
*  参数说明：无
*  函数返回：无
*  修改时间：2020年3月10日
*  备    注：
*************************************************************************/
int core0_main (void)
{

    CPU_Init();     // 芯片初始化

    GPIO_LED_Init();    // LED初始化

    TFTSPI_Init(1);        //LCD初始化  0:横屏  1：竖屏
    TFTSPI_CLS(u16BLUE);   //蓝色屏幕
    TFTSPI_P8X16Str(0, 0, "Test Encoder",u16WHITE,u16BLACK);      //字符串显示

    Servos_Io_init();       // 舵机初始化

    Mortor_Init();      // 电机初始化

	Encoder_Io_Init();  // 编码器初始化

	Morter_Set_V();     // 速度设置

	Servos_Set_Angle();     //设置舵机角度

	// 中断太快的话  导致编码器不能更新数据
	STM_InitConfig(STM0, STM_Channel_0, 10000);     // 定时器STM0_0  10000us  0.01s进入一次中断

	// 通知CPU1，CPU0初始化完成
	IfxCpu_releaseMutex(&mutexCpu0InitIsOk);
	// 切记CPU0,CPU1...不可以同时开启屏幕显示，否则冲突不显示
	mutexCpu0TFTIsOk=0;         // CPU1： 0占用/1释放 TFT

	while (1) //主循环
	{

	}

}

在这其实是有一个问题需要我们思考的，那就是，电机，舵机，还有编码器，究竟要不要放到定时器里，有没有必要都放进去，定时器对编码器来说存在一定的误差，通过测试得到定时器执行间隔太快的话，好像是测不到编码器数值的，也有可能是TFT还没刷新，但是下一次定时器中断已经开始了，直接将数据覆盖了，一直这么循环下去，造成这种屏幕数字不变的情况，在加长时间之后确实是可以测到编码器的数值了，但是它应该是会丢掉一部分的反馈的。

至于该怎么处理这个问题，暂时还没想出来。但是这样应该不耽误使用，对于智能车来说应该是足够了。
