/**
 * @brief 求2个数的绝对值
 *
 * @param x 参数1
 * @param y	参数2
 * @return float 	是否可以传输
 */
int my_abs(float x, float y)
{
	if (x > y)
	{
		if (x - y >= 1.0f)
			return 0;
		else
			return 1;
	}
	else if (y > x)
	{
		if (y - x >= 1.0f)
			return 0;
		else
			return 1;
	}
	else if (x == y)
		return 1;

	return 1;
}

int main(void)
{

	NVIC_PriorityGroupConfig(NVIC_PriorityGroup_4); // 设置系统中断优先级分组4
	delay_init(168);								// 初始化延时函数
	csb1_IO_init();
	csb2_IO_init();
	uart_init(115200);
	Tim3_Init();
	Tim4_Init();
	TIM2_PWM_Init(2000 - 1, 840 - 1); // T = 84M / (ARR + 1) / (PSC + 1) = 20ms
	Motor_init_Io();

	while (HC05_Init()) // 初始化ATK-HC05模块
	{
		delay_ms(200);
	}

	USART3_RX_STA = 0;

	delay_ms(1200); // 等待步进电机上电完成

	// 创建开始任务
	xTaskCreate((TaskFunction_t)start_task,			 // 任务函数
				(const char *)"start_task",			 // 任务名称
				(uint16_t)START_STK_SIZE,			 // 任务堆栈大小
				(void *)NULL,						 // 传递给任务函数的参数
				(UBaseType_t)START_TASK_PRIO,		 // 任务优先级
				(TaskHandle_t *)&StartTask_Handler); // 任务句柄
	vTaskStartScheduler();							 // 开启任务调度
}

// 开始任务任务函数
void start_task(void *pvParameters)
{
	taskENTER_CRITICAL(); // 进入临界区
	// 创建LED0任务
	xTaskCreate((TaskFunction_t)led0_task,
				(const char *)"led0_task",
				(uint16_t)LED0_STK_SIZE,
				(void *)NULL,
				(UBaseType_t)LED0_TASK_PRIO,
				(TaskHandle_t *)&LED0Task_Handler);
	// 创建LED1任务
	xTaskCreate((TaskFunction_t)led1_task,
				(const char *)"led1_task",
				(uint16_t)LED1_STK_SIZE,
				(void *)NULL,
				(UBaseType_t)LED1_TASK_PRIO,
				(TaskHandle_t *)&LED1Task_Handler);
	// 浮点测试任务
	xTaskCreate((TaskFunction_t)float_task,
				(const char *)"float_task",
				(uint16_t)FLOAT_STK_SIZE,
				(void *)NULL,
				(UBaseType_t)FLOAT_TASK_PRIO,
				(TaskHandle_t *)&FLOATTask_Handler);
	vTaskDelete(StartTask_Handler); // 删除开始任务
	taskEXIT_CRITICAL();			// 退出临界区
}

/**
 * @brief 超声波控制部分
 *
 * @param pvParameters
 */
void led0_task(void *pvParameters)
{

	while (1)
	{
		float juli = 0;
		float juli2 = 0;
		float juli_end = 0;
		float zuobiao[3] = {0}; // x y z  接收端 村的数据库是 z y x 有点差异

		juli = Get_CSB1_Value();
		vTaskDelay(50);
		juli2 = Get_CSB2_Value();
		vTaskDelay(50);

		if (juli >= 2 && juli <= 400 && juli2 >= 2 && juli2 <= 400)
		{
			if (my_abs(juli, juli2))
			{
				juli_end = (juli + juli2) / 2.0f;

				zuobiao[0] = juli_end * cos(y) * cos(x); // x 坐标

				zuobiao[1] = juli_end * cos(y) * sin(x); // y 坐标

				zuobiao[2] = juli_end * sin(y); // z 坐标

				u3_printf("%f,%f,%f,juli=%f, x=%d,y=%f\r\n", zuobiao[0], zuobiao[1], zuobiao[2], juli_end, x, y); // 发送到蓝牙模块
			}
		}
	}
}

/**
 * @brief 电机控制任务		越大方波越小
 *
 * @param pvParameters

void led1_task(void *pvParameters)

	}
}

// 定义成全局变量
// float x = 0; // 水平面上转过的角度

// float y = 0; // 竖直面上转过的角度

void float_task(void *pvParameters)
{

	int t = 1950;
	int dir = 1; // 开始时按正方向转

	while (1)
	{
		if (dir == 1)
		{ // 正方向  这要考虑 电机的角度问题 是 0 ---> 360 还是 360 ----> 0 这块会影响计算
			for (t = 1950; t >= 1750; t--)
			{
				TIM_SetCompare1(TIM2, t);
				vTaskDelay(1000);
				y += 0.9; // 舵机
				if (t == 1750)
				{
					dir = 0;	 // 舵机方向改变
					flag_bj = 1; // 通知步进电机可以转了
				}
			}
		}
		else if (dir == 0)
		{ // 反方向
			for (t = 1750; t <= 1950; t++)
			{
				TIM_SetCompare1(TIM2, t);
				vTaskDelay(1000);
				y -= 0.9;
				if (t == 1950)
				{
					dir = 1;	 // 舵机方向改变
					flag_bj = 1; // 通知步进电机可以转了
				}
			}
		}
	}
}
